Printing

  • Printing messages
    • printf("This is message \n");
  • Printing variables
    • printf("format", parameters);

int i = 20;
char c = 'a';
printf("%d, %c", i, c);
printf("i is %d and char is %c", i, '6');

Printing Integers

  • %d, %i, %ld

printf("%d", 100);
100
printf("%d, %d", +1000, -100);
1000, -100
printf("%i", 100);
100
printf("%ld, %i", +1000, -100);
1000, -100

Printing Unsigned Integers

  • %u (base 10), %o (base 8), %x (base 16) and %X (base 16)

unsigned int i = 26;
printf("%u\n", i); //26
printf("%o\n", i); //32
printf("%x\n", i); //1a
printf("%X\n", i); //1A

Printing Floats

  • %f, %e, %E, %lf

printf("%f", 100.5f);
100.500000

float f = -2;
double d = 100;
printf("%f, %f", f, d);
-2.000000, 100.000000
printf("%f, %e", 1e3, 1e3);
1000.000000, 1.000000e+003

Printing Chars

  • %c

printf("%c", 'a');
a

printf("%c, %c", 'a','b');
a, b

char c1 = 'a';
printf("%c, %c, %c", c1, 'b', 65);
a, b, A
