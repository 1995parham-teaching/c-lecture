Arrays in C

  • Array declaration in C

    • <Elements’ Type> <identifier> [ <size> ]

  • <Elements’ Type> : int, char, float, ...
  • <size> :
    • Old compilers (standard): it should be constant
    • New compilers (standard): it can be variable
  • Elements in array
    • From 0 to (size – 1)

Out-of-range access

  • C compiler does not check the range you access
      • 
int x[10]; x[20] = 30; y = x[100];
    • No compiler error!
  • What happen
    • Read: Logical error
      • 
y = x[100];
    • Write: May or may not logical or runtime error
      • 
x[20] = 30;

#include <stdio.h>

void init_array(int size, int arr[]){
  int i;
  for(i = 0; i < size; i++)
    arr[i] = i;
}

  int i;
  for(i = 0; i < size; i++)
    printf("arr[%d] = %d\n", i, arr[i]);

Max Index Example


#include <stdio.h>

int max_index(int a[], int size){
  int i;
  int index = 0;
  for(i = 1; i < size; i++)
    if(a[i] > a[index])
      index = i;
  return index;
}

void main(void){
  int arr[] = {1, 4, 12, 93, 23};

Sort


#include <stdio.h>

void array_swap(int a[], int i, int j){
  int tmp;
  tmp = a[i];
  a[i] = a[j];
  a[j] = tmp;
}

void bubble_sort(int a[], int size){
  int i, j;

  for(i = 0; i < size - 1; i++)
    for(j = i + 1; j < size; j++)
      if(a[i] < a[j])
        array_swap(a, i, j);
}

void print(int a[], int size){
  int i;
  for(i = 0; i < size; i++)
    printf("%d ", a[i]);
  printf("\n");
}

int main(void){
  int arr[] = {1, 7, 3, 7, 11, 0};
  int size = sizeof(arr) / sizeof(arr[0]);

  printf("Before sort: ");
  print(arr, size);

  bubble_sort(arr, size);

  printf("After sort: ");
  print(arr, size);

  return 0;
}

Binary Search

  • Given a sorted array arr[] of n elements, write a function to search a given element x in arr[].

int bsearch(int start, int end, int a[], int value){
  int mid = (start + end) / 2;

  if(a[mid] == value)
     return mid;
  else if(start >= end)
    return -1;
  else if(value > a[mid])
    return bsearch(mid + 1, end, a, value);
  else
    return bsearch(start, mid - 1 , a, value); }

Matrix Transpose


#define SIZE 5

void swap(int a[SIZE][SIZE], int i, int j){
  int tmp;
  tmp = a[i][j];
  a[i][j] = a[j][i];
  a[j][i] = tmp;
}

void transpose(int a[][SIZE]){
  int i, j;
  for(i = 0; i < SIZE; i++)
    for(j = i; j < SIZE; j++)
      swap(a, i, j);
}
