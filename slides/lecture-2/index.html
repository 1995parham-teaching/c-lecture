Factorial-1


print “Please enter a positive integer”
read n
i = 1
result = 1
while (i <= n)
  result = i * result
  i = i + 1
return result

Factorial-2


print “Please enter a positive integer”
read n
result = 1
while (n > 0)
  result = result * n
  n = n - 1
return result

Factorial-Recursive (n)


if (n == 1)
  return 1
else
  return n * Factorial-Recursive (n - 1)

sort (x, start, end)


while (start != end)
  j = find index of minimum element from start to end
  swap x[j] and x[start]
  start = start + 1

find_min(x, start, end)


i = start
arg_min = i
while (i <= end)
  if(x[i] < x[arg_min])
    arg_min = i
  i = i + 1
return arg_min

swap(x, j, i)


temp = x[j]
x[j] = x[i]
x[i] = temp

reverse(A, start, end)


if (start >= end)
  return
else
  swap(A, start, end)
  reverse(A, start + 1, end - 1)
