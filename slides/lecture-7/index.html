Repetition

  • Repetition is performed by loops
    • Put all statements to repeat in a loop
  • Don’t loop to infinity
    • Stop the repetition
    • Based on some conditions (counter, sentinel)
  • C has three statements for loops
    • while statement
    • do-while statement
    • for statement

while statement

while ( <expression> ) <statements>

Print 0 to n


#include <stdio.h>
int main(void){
  int n, number;
  number = 0;
  printf("Enter n: ");
  scanf("%d", &n);
  while(number <= n) {
    printf("%d \n", number);
    number++;
  }
  return 0;
}

do-while statement

<statements>
while ( <expression> );

Sum of the n term: 1/2 + 2/3 ...


#include <stdio.h>
#include <stdlib.h>

int main(void) {
  int n;
  double number, sum;

  printf("Enter n > 0: ");
  scanf("%d", &n);

  if(n < 1){
    printf("wrong input");
    return -1;
  }

  sum = 0;
  number = 0.0;
  do {
    number++;
    sum += number / (number + 1.0);
  } while(number < n);
  
  printf("sum = %f\n", sum);
  return 0;
}
